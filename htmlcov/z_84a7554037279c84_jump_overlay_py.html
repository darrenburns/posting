45 def on_key(self, key_event: events.Key) -> None: 

46 # We need to stop the bubbling of these keys, because if they 

47 # arrive at the parent after the overlay is closed, then the parent 

48 # will handle the key event, resulting in the focus being shifted 

49 # again (unexpectedly) after the jump target was focused. 

50 if key_event.key == "tab" or key_event.key == "shift+tab": 

51 key_event.stop() 

52 key_event.prevent_default()

55 # If they press a key corresponding to a jump target, 

56 # then we jump to it. 

57 target = self.keys_to_widgets.get(key_event.key) 

58 if target is not None: 

59 self.dismiss(target) 

60 return
