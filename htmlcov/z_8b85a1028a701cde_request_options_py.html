143 @on(DescendantFocus) 

144 def on_descendant_focus(self, event: DescendantFocus) -> None: 

145 """Show the description of the option when the user focuses on it.""" 

146 focused_id = event.control.id 

147 description_label = self.query_one("#option-description", Static) 

148 has_description = focused_id in self.descriptions 

149 description_label.set_class(has_description, "show") 

150 if has_description: 

151 description_label.update(self.descriptions[focused_id]) 

152 else: 

153 description_label.update("")
