29 def to_httpx_auth(self) -> httpx.Auth | None: 

30 if self.type == "basic": 

31 assert self.basic is not None 

32 return httpx.BasicAuth(self.basic.username, self.basic.password) 

33 elif self.type == "digest": 

34 assert self.digest is not None 

35 return httpx.DigestAuth(self.digest.username, self.digest.password)

39 return None

197 def apply_template(self, variables: dict[str, Any]) -> None: 

198 """Apply the template to the request model.""" 

199 try: 

200 template = Template(self.url) 

201 self.url = ensure_protocol(template.substitute(variables)) 

202 

203 template = Template(self.description) 

204 self.description = template.substitute(variables) 

205 template = Template(self.options.proxy_url) 

206 self.options.proxy_url = template.substitute(variables) 

207 

208 if self.body: 

209 if self.body.content: 

210 template = Template(self.body.content) 

211 self.body.content = template.substitute(variables) 

212 if self.body.form_data: 

213 for item in self.body.form_data: 

214 template = Template(item.name) 

215 item.name = template.substitute(variables) 

216 template = Template(item.value) 

217 item.value = template.substitute(variables) 

218 

219 for header in self.headers: 

220 template = Template(header.name) 

221 header.name = template.substitute(variables) 

222 template = Template(header.value) 

223 header.value = template.substitute(variables) 

224 for param in self.params: 

225 template = Template(param.name) 

226 param.name = template.substitute(variables) 

227 template = Template(param.value) 

228 param.value = template.substitute(variables) 

229 

230 if self.auth is not None: 

231 if self.auth.basic is not None: 

232 template = Template(self.auth.basic.username) 

233 self.auth.basic.username = template.substitute(variables) 

234 template = Template(self.auth.basic.password) 

235 self.auth.basic.password = template.substitute(variables) 

236 if self.auth.digest is not None: 

237 template = Template(self.auth.digest.username) 

238 self.auth.digest.username = template.substitute(variables) 

239 template = Template(self.auth.digest.password)

241 if self.auth.bearer_token is not None: 

242 template = Template(self.auth.bearer_token.token)

290 def to_curl(self, extra_args: str = "") -> str: 

291 """Convert the request model to a cURL command. 

292 

293 Optionally supply extra arguments to insert into the command. 

294 

295 Args: 

296 extra_args: A string of extra arguments to insert into the command. 

297 

298 Returns: 

299 A string of the cURL command that can be used via the command line. 

300 """ 

301 parts = ["curl"] 

302 if extra_args: 

303 parts.append(extra_args) 

304 

305 if self.method != "GET": 

306 parts.append(f"-X {self.method}") 

307 

308 for header in self.headers: 

309 if header.enabled: 

310 parts.append(f"-H '{header.name}: {header.value}'") 

311 

312 parsed_url = urlparse(self.url) 

313 existing_params = parse_qsl(parsed_url.query) 

314 if self.params: 

315 for param in self.params: 

316 if param.enabled: 

317 existing_params.append((param.name, param.value)) 

318 

319 new_query = urlencode(existing_params) 

320 new_url = urlunparse( 

321 ( 

322 parsed_url.scheme, 

323 parsed_url.netloc, 

324 parsed_url.path, 

325 parsed_url.params, 

326 new_query, 

327 parsed_url.fragment, 

328 ) 

329 ) 

330 

331 if self.body and self.body.content: 

332 parts.append(f"-d '{self.body.content}'") 

333 

334 if self.body and self.body.form_data: 

335 for item in self.body.form_data: 

336 if item.enabled: 

337 parts.append(f"-d '{item.name}={item.value}'") 

338 

339 if self.auth: 

340 if self.auth.type == "basic" and self.auth.basic: 

341 parts.append( 

342 f"-u '{self.auth.basic.username}:{self.auth.basic.password}'" 

343 ) 

344 elif self.auth.type == "digest" and self.auth.digest: 

345 parts.append( 

346 f"--digest -u '{self.auth.digest.username}:{self.auth.digest.password}'" 

347 ) 

348 

349 for cookie in self.cookies: 

350 if cookie.enabled: 

351 parts.append(f"--cookie '{cookie.name}={cookie.value}'") 

352 

353 if not self.options.follow_redirects: 

354 parts.append("--no-location") 

355 

356 if not self.options.verify_ssl: 

357 parts.append("--insecure") 

358 

359 if self.options.timeout != 5.0: # Only add if not the default 

360 parts.append(f"--max-time {self.options.timeout}") 

361 

362 if self.options.proxy_url: 

363 parts.append(f"--proxy '{self.options.proxy_url}'") 

364 

365 parts.append(f"'{new_url}'") 

366 

367 # Join with newlines for better readability, similar to other tools 

368 return " \\\n ".join(parts)

459 def from_directory(cls, directory: str) -> Collection: 

460 """Load all request models into a tree structure from a directory containing .posting.yaml files. 

461 

462 Args: 

463 directory_path: The path to the directory containing .posting.yaml files. 

464 

465 Returns: 

466 Collection: The root collection containing all loaded requests and subcollections. 

467 """ 

468 if directory: 

469 directory_path = Path(directory) 

470 else: 

471 directory_path = Path.cwd() 

472 directory = str(directory_path) 

473 

474 request_files = directory_path.rglob("*.posting.yaml") 

475 collection_name = directory_path.name 

476 root_collection = Collection(name=collection_name, path=directory_path) 

477 

478 for file_path in request_files: 

479 try: 

480 path_string = str(file_path) 

481 request = load_request_from_yaml(path_string) 

482 path_parts = ( 

483 path_string[len(directory) :].strip(os.path.sep).split(os.path.sep) 

484 ) 

485 current_level = root_collection 

486 subpath = root_collection.path 

487 for part in path_parts[:-1]: 

488 subpath = subpath / part 

489 found = False 

490 for child in current_level.children: 

491 if child.name == part: 

492 current_level = child 

493 found = True 

494 break 

495 

496 if not found: 

497 new_collection = Collection(name=part, path=subpath) 

498 current_level.children.append(new_collection) 

499 current_level = new_collection 

500 current_level.requests.append(request) 

501 except Exception as e: 

502 print(f"Failed to load {file_path}: {e}") 

503 

504 # Sort the requests and children at all levels of the tree 

505 def sort_collection(collection: Collection): 

506 # Sort subcollections 

507 collection.children.sort(key=lambda x: x.name) 

508 

509 # Sort requests in this collection 

510 collection.requests.sort(key=request_sort_key) 

511 

512 # Recursively sort child collections 

513 for child in collection.children: 

514 sort_collection(child) 

515 

516 sort_collection(root_collection) 

517 return root_collection
