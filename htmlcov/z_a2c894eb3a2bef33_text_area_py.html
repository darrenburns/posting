264 def on_key(self, event: events.Key) -> None: 

265 character = event.character 

266 

267 if self.read_only: 

268 return 

269 

270 if character in self.OPENING_BRACKETS: 

271 opener = character 

272 closer = self.OPENING_BRACKETS[opener] 

273 self.insert(f"{opener}{closer}") 

274 self.move_cursor_relative(columns=-1) 

275 event.prevent_default() 

276 elif character in self.CLOSING_BRACKETS: 

277 # If we're currently at a closing bracket and 

278 # we type the same closing bracket, move the cursor 

279 # instead of inserting a character. 

280 if self._matching_bracket_location: 

281 row, col = self.cursor_location 

282 line = self.document.get_line(row) 

283 if character == line[col]: 

284 event.prevent_default() 

285 self.move_cursor_relative(columns=1) 

286 elif event.key == "enter": 

287 row, column = self.cursor_location 

288 line = self.document.get_line(row) 

289 if not line: 

290 return 

291 

292 column = min(column, len(line) - 1) 

293 character_locations = self._yield_character_locations_reverse( 

294 (row, max(0, column - 1)) 

295 ) 

296 rstrip_line = line[: column + 1].rstrip() 

297 anchor_char = rstrip_line[-1] if rstrip_line else None 

298 get_content_start_column = self.get_content_start_column 

299 get_column_width = self.get_column_width 

300 try: 

301 for character, _location in character_locations: 

302 # Ignore whitespace 

303 if character.isspace(): 

304 continue 

305 elif character in self.OPENING_BRACKETS: 

306 # We found an opening bracket on this line, 

307 # so check the indentation of the line. 

308 # The newly created line should have increased 

309 # indentation. 

310 content_start_col = get_content_start_column(line) 

311 width = get_column_width(row, content_start_col) 

312 width_to_indent = max( 

313 width + self.indent_width, self.indent_width 

314 ) 

315 

316 target_location = row + 1, column + width_to_indent 

317 insert_text = "\n" + " " * width_to_indent 

318 if anchor_char in self.CLOSING_BRACKETS: 

319 # If there's a bracket under the cursor, we should 

320 # ensure that gets indented too. 

321 insert_text += "\n" + " " * content_start_col 

322 

323 self.insert(insert_text) 

324 self.cursor_location = target_location 

325 event.prevent_default() 

326 break 

327 else: 

328 content_start_col = get_content_start_column(line) 

329 width = get_column_width(row, content_start_col) 

330 self.insert("\n" + " " * width) 

331 event.prevent_default() 

332 break 

333 except IndexError: 

334 return 

335 

336 self._restart_blink()

346 def _yield_character_locations_reverse( 

347 self, start: Location 

348 ) -> Iterable[tuple[str, Location]]: 

349 row, column = start 

350 document = self.document 

351 line_count = document.line_count 

352 

353 while line_count > row >= 0: 

354 line = document[row] 

355 if column == -1: 

356 column = len(line) - 1 

357 while column >= 0: 

358 yield line[column], (row, column) 

359 column -= 1
