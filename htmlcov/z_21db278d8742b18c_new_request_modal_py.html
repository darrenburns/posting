44class DirectoryValidator(Validator): 

45 def validate(self, value: str) -> ValidationResult: 

46 def is_valid_directory(value: str) -> bool: 

47 if not value or ".." in value or value.startswith("/") or ":" in value: 

48 return False 

49 return True 

50 

51 return ( 

52 self.success() 

53 if is_valid_directory(value)
