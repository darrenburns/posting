168def content_type_to_language(content_type: str) -> str | None: 

169 """Given the value of an HTTP content-type header, return the name 

170 of the language to use in the response body text area.""" 

171 if content_type.startswith("application/json"): 

172 return "json" 

173 elif content_type.startswith("text/html") or content_type.startswith( 

174 "application/xml" 

175 ): 

176 return "html" 

177 elif content_type.startswith("text/css"): 

178 return "css" 

179 elif content_type.startswith("text/plain"): 

180 return None 

181 return "json"
