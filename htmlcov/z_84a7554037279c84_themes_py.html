49 def to_text_area_syntax_styles(self, fallback_theme: "Theme") -> dict[str, Style]: 

50 """Convert this theme to a TextAreaTheme. 

51 

52 If a fallback theme is provided, it will be used to fill in any missing 

53 styles. 

54 """ 

55 syntax_styles = { 

56 "string": Style.parse(self.json_string or fallback_theme.primary), 

57 "number": Style.parse(self.json_number or fallback_theme.accent), 

58 "boolean": Style.parse(self.json_boolean or fallback_theme.accent), 

59 "json.null": Style.parse(self.json_null or fallback_theme.secondary), 

60 "json.label": ( 

61 Style.parse(self.json_key or fallback_theme.primary) + Style(bold=True) 

62 ), 

63 } 

64 return syntax_styles

170 def to_text_area_theme(self) -> TextAreaTheme: 

171 """Retrieve the TextAreaTheme corresponding to this theme.""" 

172 syntax_styles: dict[str, Style] = {} 

173 if isinstance(self.syntax, SyntaxTheme): 

174 syntax_styles = self.syntax.to_text_area_syntax_styles(self) 

175 

176 text_area = self.text_area 

177 return TextAreaTheme( 

178 name=uuid.uuid4().hex, 

179 syntax_styles=syntax_styles, 

180 gutter_style=Style.parse(text_area.gutter) if text_area.gutter else None, 

181 cursor_style=Style.parse(text_area.cursor) if text_area.cursor else None, 

182 cursor_line_style=Style.parse(text_area.cursor_line) 

183 if text_area.cursor_line 

184 else None, 

185 cursor_line_gutter_style=Style.parse(text_area.cursor_line_gutter) 

186 if text_area.cursor_line_gutter 

187 else None, 

188 bracket_matching_style=Style.parse(text_area.matched_bracket) 

189 if text_area.matched_bracket 

190 else None, 

191 selection_style=Style.parse(text_area.selection) 

192 if text_area.selection

197def load_user_themes() -> dict[str, Theme]: 

198 """Load user themes from "~/.config/posting/themes". 

199 

200 Returns: 

201 A dictionary mapping theme names to theme objects. 

202 """ 

203 directory = SETTINGS.get().theme_directory 

204 themes: dict[str, Theme] = {} 

205 for path in directory.iterdir(): 

206 path_suffix = path.suffix 

207 if path_suffix == ".yaml" or path_suffix == ".yml": 

208 with path.open() as theme_file: 

209 theme_content = yaml.load(theme_file, Loader=yaml.FullLoader) or {} 

210 try: 

211 themes[theme_content["name"]] = Theme(**theme_content) 

212 except KeyError: 

213 raise ValueError( 

214 f"Invalid theme file {path}. A `name` is required." 

215 ) 

216 return themes
