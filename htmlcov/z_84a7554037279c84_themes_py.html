34def load_user_themes() -> dict[str, Theme]: 

35 """Load user themes from "~/.config/posting/themes".""" 

36 directory = SETTINGS.get().theme_directory 

37 themes: dict[str, Theme] = {} 

38 for path in directory.iterdir(): 

39 path_suffix = path.suffix 

40 if path_suffix == ".yaml" or path_suffix == ".yml": 

41 with path.open() as theme_file: 

42 theme_content = yaml.load(theme_file, Loader=yaml.FullLoader) or {} 

43 try: 

44 themes[theme_content["name"]] = Theme(**theme_content) 

45 except KeyError: 

46 raise ValueError( 

47 f"Invalid theme file {path}. A `name` is required." 

48 ) 

49 return themes
