314def load_user_themes() -> UserThemeLoadResult: 

315 """Load user themes from the theme directory. 

316 

317 The theme directory is defined in the settings file as `theme_directory`. 

318 

319 You can locate it on the command line with `posting locate themes`. 

320 

321 Returns: 

322 A dictionary mapping theme names to theme objects. 

323 """ 

324 directory = SETTINGS.get().theme_directory 

325 themes: dict[str, TextualTheme] = {} 

326 failed_themes: list[tuple[Path, Exception]] = [] 

327 

328 for path in directory.iterdir(): 

329 path_suffix = path.suffix 

330 if path_suffix == ".yaml" or path_suffix == ".yml": 

331 try: 

332 theme = load_user_theme(path) 

333 if theme: 

334 themes[theme.name] = theme 

335 except Exception as e: 

336 failed_themes.append((path, e)) 

337 

338 return UserThemeLoadResult(loaded_themes=themes, failed_themes=failed_themes)

341def load_user_theme(path: Path) -> TextualTheme | None: 

342 with path.open() as theme_file: 

343 try: 

344 theme_content = yaml.load(theme_file, Loader=yaml.FullLoader) or {} 

345 except Exception as e: 

346 raise InvalidThemeError(f"Could not parse theme file: {str(e)}.") 

347 

348 try: 

349 return Theme(**theme_content).to_textual_theme() 

350 except Exception: 

351 raise InvalidThemeError(f"Invalid theme file at {str(path)}.")
